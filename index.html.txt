function loadQuestion() {
  if (qIdx >= questions.length) { showResult(true); return; }
  const q = questions[qIdx];
  document.getElementById('battle-count').textContent = `${qIdx+1}/${questions.length}`;
  document.getElementById('q-type').textContent = q.type;
  document.getElementById('q-text').innerHTML = q.q.replace('___', '<span class="q-blank">___</span>');

  const monsters = ['答對才能前進！', '你敢挑戰我嗎？', '這題不簡單喔...', '看看你的實力！'];
  document.getElementById('monster-dialog').textContent = monsters[qIdx % monsters.length];

  const grid = document.getElementById('answer-grid');
  grid.innerHTML = '';
  q.opts.forEach((opt, i) => {
    const btn = document.createElement('button');
    btn.className = 'answer-btn';
    btn.textContent = opt;
    btn.onclick = () => answerQ(i, q.ans, btn, q);
    grid.appendChild(btn);
  });
}

function answerQ(chosen, correct, btn, q) {
  const btns = document.querySelectorAll('.answer-btn');
  btns.forEach(b => b.onclick = null);

  if (chosen === correct) {
    btn.classList.add('correct');
    combo++;
    enemyHP = Math.max(0, enemyHP - (20 + combo * 5));
    showDamage(`⚔️ -${20 + combo * 5}`, 'var(--green)');
    document.getElementById('monster-dialog').textContent = q.explain + ' ✅';
    showCombo();
  } else {
    btn.classList.add('wrong');
    btns[correct].classList.add('correct');
    combo = 0;
    playerHP = Math.max(0, playerHP - 20);
    showDamage('💔 -20', 'var(--red)');
    document.getElementById('monster-dialog').textContent = q.explain + ' ❌';
    hideCombo();
  }

  updateBattleHUD();

  if (playerHP <= 0) { setTimeout(() => showResult(false), 1000); return; }
  if (enemyHP <= 0) { setTimeout(() => showResult(true), 1000); return; }

  setTimeout(() => { qIdx++; loadQuestion(); }, 1200);
}

  document.getElementById('enemy-hp-bar').style.width = enemyHP + '%';
  document.getElementById('enemy-hp-val').textContent = `${enemyHP} / 100`;
}

  el.style.cssText = `color:${color};top:200px;left:${40 + Math.random()*20}%;font-family:var(--pixel);font-size:14px;`;
  el.textContent = text;
  document.body.appendChild(el);
  setTimeout(() => el.remove(), 1000);
}

function showCombo() {
  if (combo < 2) return;
  const bar = document.getElementById('combo-bar');
  document.getElementById('combo-num').textContent = `${combo}x`;
  bar.classList.add('show');
  clearTimeout(window._comboTimer);
  window._comboTimer = setTimeout(() => bar.classList.remove('show'), 2000);
}
function hideCombo() { document.getElementById('combo-bar').classList.remove('show'); }

function showResult(win) {
  const overlay = document.getElementById('result-overlay');
  document.getElementById('result-emoji').textContent = win ? '🏆' : '💀';
  document.getElementById('result-title').textContent = win ? 'STAGE CLEAR!' : 'GAME OVER';
  document.getElementById('result-desc').textContent = win
    ? `太強了！擊敗了 ${ZONES[currentZone].monsterName}！連擊最高 ${combo}x` 
    : '不要灰心！重新挑戰，你可以的！';

  document.getElementById('reward-xp').textContent = `+${xp} XP`;
  document.getElementById('reward-gold').textContent = win ? `+${50 + combo * 5} 🪙` : '+5 🪙';

}

// ═══════════ VOCAB ═══════════
function loadVocab() {
  const v = VOCABS[vocabIdx];
  document.getElementById('v-level').textContent = v.level;
  document.getElementById('v-cat').textContent = v.cat;
  document.getElementById('v-word').textContent = v.word;
  document.getElementById('v-phone').textContent = v.phone;
  document.getElementById('v-zh').textContent = v.zh;
  document.getElementById('v-example').textContent = v.ex;
  document.getElementById('v-example').classList.remove('show');
  document.getElementById('vocab-counter').textContent = `${vocabIdx+1}/${VOCABS.length}`;
  vocabFlipped = false;
  loadQuickQuiz(v);
}

function flipVocab() {
  vocabFlipped = !vocabFlipped;
  document.getElementById('v-example').classList.toggle('show', vocabFlipped);
}

function nextVocab(rating) {
  vocabIdx = (vocabIdx + 1) % VOCABS.length;
  loadVocab();
}

function loadQuickQuiz(v) {
  document.getElementById('quick-q').textContent = `「${v.zh}」的英文是？`;
  const opts = [v.word, VOCABS[(vocabIdx+1)%VOCABS.length].word, VOCABS[(vocabIdx+2)%VOCABS.length].word, VOCABS[(vocabIdx+3)%VOCABS.length].word];
  const shuffled = opts.sort(() => Math.random() - 0.5);
  const container = document.getElementById('quick-opts');
  container.innerHTML = '';
  shuffled.forEach(opt => {
    const btn = document.createElement('button');
    btn.style.cssText = 'background:var(--card);border:1px solid var(--border2);border-radius:10px;padding:10px 14px;font-family:var(--body);font-size:13px;color:var(--text);text-align:left;cursor:pointer;transition:all 0.15s;width:100%';
    btn.textContent = opt;
    btn.onclick = () => {
      const correct = opt === v.word;
      btn.style.borderColor = correct ? 'var(--green)' : 'var(--red)';
      btn.style.color = correct ? 'var(--green)' : 'var(--red)';
      container.querySelectorAll('button').forEach(b => b.onclick = null);
      if (correct) showDamage('⚡ +20 XP', 'var(--gold)');
    };
    container.appendChild(btn);
  });
}
